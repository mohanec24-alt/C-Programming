struct Node* insertEnd(struct Node* head, int data) 
{
    struct Node* newNode = createNode(data);
    if (head == NULL)
        return newNode;
    struct Node* temp = head;
    while (temp->next != NULL)
        temp = temp->next;
    temp->next = newNode;
    newNode->prev = temp;
    return head;
}

{
    struct Node* newNode = createNode(data);
    if (pos == 1)

        if (head != NULL)
            head->prev = newNode;
        return newNode;

}
